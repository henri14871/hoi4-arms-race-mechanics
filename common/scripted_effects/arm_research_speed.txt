# Arms Race Mechanics - Research Speed Spirit Swap Logic
# Scripted effect: arm_update_research_speed_spirit
# Runs in COUNTRY scope. Called during each evaluation cycle.

arm_clear_tier_research_spirits = {
    if = { limit = { has_idea = arm_tier_superpower } remove_ideas = arm_tier_superpower }
    if = { limit = { has_idea = arm_tier_great_power } remove_ideas = arm_tier_great_power }
    if = { limit = { has_idea = arm_tier_regional_power } remove_ideas = arm_tier_regional_power }
    if = { limit = { has_idea = arm_tier_minor_industrial } remove_ideas = arm_tier_minor_industrial }
    if = { limit = { has_idea = arm_tier_minor } remove_ideas = arm_tier_minor }
    if = { limit = { has_idea = arm_tier_micro } remove_ideas = arm_tier_micro }
}

arm_apply_tier_research_spirit = {
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 5 compare = equals }
            OR = {
                NOT = { has_idea = arm_tier_superpower }
                has_idea = arm_tier_great_power
                has_idea = arm_tier_regional_power
                has_idea = arm_tier_minor_industrial
                has_idea = arm_tier_minor
                has_idea = arm_tier_micro
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_superpower
    }
    else_if = {
        limit = {
            check_variable = { var = arm_tier_index value = 4 compare = equals }
            OR = {
                NOT = { has_idea = arm_tier_great_power }
                has_idea = arm_tier_superpower
                has_idea = arm_tier_regional_power
                has_idea = arm_tier_minor_industrial
                has_idea = arm_tier_minor
                has_idea = arm_tier_micro
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_great_power
    }
    else_if = {
        limit = {
            check_variable = { var = arm_tier_index value = 3 compare = equals }
            OR = {
                NOT = { has_idea = arm_tier_regional_power }
                has_idea = arm_tier_superpower
                has_idea = arm_tier_great_power
                has_idea = arm_tier_minor_industrial
                has_idea = arm_tier_minor
                has_idea = arm_tier_micro
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_regional_power
    }
    else_if = {
        limit = {
            check_variable = { var = arm_tier_index value = 2 compare = equals }
            OR = {
                NOT = { has_idea = arm_tier_minor_industrial }
                has_idea = arm_tier_superpower
                has_idea = arm_tier_great_power
                has_idea = arm_tier_regional_power
                has_idea = arm_tier_minor
                has_idea = arm_tier_micro
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_minor_industrial
    }
    else_if = {
        limit = {
            check_variable = { var = arm_tier_index value = 1 compare = equals }
            OR = {
                NOT = { has_idea = arm_tier_minor }
                has_idea = arm_tier_superpower
                has_idea = arm_tier_great_power
                has_idea = arm_tier_regional_power
                has_idea = arm_tier_minor_industrial
                has_idea = arm_tier_micro
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_minor
    }
    else_if = {
        limit = {
            OR = {
                NOT = { has_idea = arm_tier_micro }
                has_idea = arm_tier_superpower
                has_idea = arm_tier_great_power
                has_idea = arm_tier_regional_power
                has_idea = arm_tier_minor_industrial
                has_idea = arm_tier_minor
            }
        }
        arm_clear_tier_research_spirits = yes
        add_ideas = arm_tier_micro
    }
}

arm_update_research_speed_spirit = {
    if = {
        limit = { has_game_rule = { rule = arm_research_speed_bonuses option = ARM_RSB_DISABLED } }
        arm_clear_tier_research_spirits = yes
        arm_clear_land_research_spirits = yes
        arm_clear_air_research_spirits = yes
        arm_clear_naval_research_spirits = yes
        arm_clear_industry_research_spirits = yes
    }
    else_if = {
        limit = { has_game_rule = { rule = arm_research_speed_bonuses option = ARM_RSB_TIER_ONLY } }
        arm_apply_tier_research_spirit = yes
        arm_clear_land_research_spirits = yes
        arm_clear_air_research_spirits = yes
        arm_clear_naval_research_spirits = yes
        arm_clear_industry_research_spirits = yes
    }
    else = {
        arm_apply_tier_research_spirit = yes
        arm_apply_land_research_spirit = yes
        arm_apply_air_research_spirit = yes
        arm_apply_naval_research_spirit = yes
        arm_apply_industry_research_spirit = yes
    }
}
